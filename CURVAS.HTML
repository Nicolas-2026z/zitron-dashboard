─── CALCULAR CURVA S ───────────────────────────────────────────────────────── def calcular_curva(nombre_pedido, nombre_proyecto, tareas): """ Devuelve dict con todos los campos que DATA necesita. """ hoy = date.today() # Separar tareas de kickoff y normales kickoffs = [t for t in tareas if t['kickoff']] normales = [t for t in tareas if not t['kickoff']] # Detectar fechas clave def min_date(lst, campo): vals = [parse_date(t[campo]) for t in lst if parse_date(t[campo])] return min(vals) if vals else None def max_date(lst, campo): vals = [parse_date(t[campo]) for t in lst if parse_date(t[campo])] return max(vals) if vals else None kickoff_date = min_date(kickoffs, 'ini') or min_date(normales, 'ini') fin_real_date = max_date(tareas, 'fin') #
